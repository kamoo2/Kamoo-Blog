## Problem

> 1 ~ N의 숫자 사이에 '+' 또는 '-' 또는 ' '을 넣어 0이 되는 수식을 찾아보자!

1. +는 더하기 -는 빼기 공백은 숫자를 이어 붙이는 것을 뜻한다.
2. 수식의 값을 계산하고 그 결과가 0이되면 출력
3. 각 테스트 케이스의 결과는 한 줄을 띄워 구분해 출력

<hr />

## Code

```javascript
let fs = require('fs');

let input = fs.readFileSync('/dev/stdin').toString().split('\n');

let T = Number(input[0]);
let N;
let arr;
let selected;

for (let t = 1; t <= T; t++) {
  // 테스트케이스 만큼 반복
  N = Number(input[t]);
  arr = Array(N)
    .fill(0)
    .map((_, index) => index + 1);
  selected = [];
  dfs(1);
  console.log();
}

function dfs(k) {
  if (k === N) {
    // N-1개의 연산자를 모두 선택한 경우
    // complete code
    let str = '';
    for (let i = 0; i < N - 1; i++) str += arr[i] + selected[i];
    str += arr[N - 1] + '';
    const result = eval(str.split(' ').join(''));
    if (result === 0) console.log(str);
    return;
  }

  for (let x of [' ', '+', '-']) {
    selected.push(x);
    dfs(k + 1);
    selected.pop();
  }
}
```
